    \begin{column}[t]{0.7\textwidth}
        Given are
        \begin{itemize}
            \item graph $G=(V,E)$
            \item one position $(n_x,n_y)\in[0,1]\times[0,1]$ for each node $n\in V$
            \item $0<r<\frac{1}{2}$
        \end{itemize}
        $G$ is called percolating, if there are nodes $n,m\in V$ such that
        \begin{enumerate}
            \item there is a cycle containing $n$ and $m$,
            \item $(n,m)\in E$,
            \item either $n_x<r$ and $m_x>1-r$ or $n_y<r$ and $m_y>1-r$.
        \end{enumerate}
    \end{column}
